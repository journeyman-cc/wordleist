031  (defn check-length
032    "Check that this `target` is exactly `word-length` long, and throuw an exception labelled
033     with `name` if it is not."
034    [target name]
035    (when-not (= (count target) word-length)
036      (throw (#?(:clj Exception.
037                 :cljs js/Error.) (str name " must have exactly " word-length " letters; '"
038                                       target "' has " (count target)))))
039    true)

046  (defn wordle
047    "given this `target` as the word sought, return the wordle pattern for this
048    `guess`."
049    [target guess]
050    (check-length target "target")
051    (check-length guess "guess")
052    (map
053     #(let [seeking (nth guess %)]
054        (cond
055          (= (nth target %) seeking) [:found seeking]
056          (some (fn [x] (= seeking x)) target) [:present seeking]
057          :else [:not-present seeking]))
058     (range (count target))))

066  (defn- in-consistent
067    "Internal guts of `consistent?`, q.v."
068    [pattern target]
069    (map #(let [present (some (fn [x] (= (nth %1 1) x)) target)]
070            (case (first %1)
071              :found (= (nth %1 1) %2)
072              :present present
073              :not-present (not present)
074              :not-found true))
075         pattern
076         target))

078  (defn consistent?
079    "Where `pattern` is a pattern as returned by `wordle`, and `target` is a
080     string representing a five letter word, return `true` it `target` is
081     consistent with `pattern`, else false."
082    [pattern target]
083    (every?
084     true?
085     (in-consistent
086      pattern
087      target)))

097  (defn ignore-not-present
098    "Replace all occurances of :not-present in this `pattern` with :not-found."
099    [pattern]
100    (map
101     #(if
102       (= :not-present (first %)) [:not-found (nth % 1)]
103       %)
104     pattern))

106  (defn refine-candidates
107    "Refine these `candidates` by removing those which are not consistent with this `pattern`,
108     generated from this `word`."
109    ([candidates pattern]
110     (refine-candidates candidates
111                        (apply str (map #(nth % 1) pattern))
112                        pattern))
113    ([candidates word pattern]
114     (filter
115      #(when (consistent?
116              (if (member? word candidates)
117                pattern
118                (ignore-not-present pattern))
119              %) %)
120      candidates)))

122  (defn pattern?
123    "Validate wordle patterns. Return `true` if `pattern` is a valid pattern, else `false`.
124  
125     A pattern is a sequence of `word-length` sequences, such that:
126     1. every element is a sequence, or nil;
127     2. the first element of every sequence which is not nil is a keyword;
128     3. the second element of every sequence which is not nil is a character."
129    [pattern]
130    (and (seq? pattern)
131         (= (count pattern) word-length)
132         (every? true?
133                 (map #(cond
134                         (nil? %) true
135                         (and (seq %) (keyword? (first %)) (char? (nth % 1))) true
136                         :else false)
137                      pattern))))

139  (defn with-mark
140    "Return a list of `word-length` elements, comprising all those characters
141     which are marked `mark` in any of these patterns."
142    [patterns mark]
143    (when-not
144     (every? pattern? patterns)
145      (throw (#?(:clj Exception.
146                 :cljs js/Error.) (str "Not a pattern: " patterns))))
147    (reduce
148     (fn [s1 s2]
149       (map #(if (char? (nth s1 %)) (nth s1 %) (nth s2 %))
150            (range word-length)))
151     (repeat word-length (vector))
152     (map
153      (fn [p]
154        (map
155         (fn [e]
156           (when (= mark (first e))
157             (nth e 1)))
158         p))
159      patterns)))

176                 p))
177              patterns))))

197  (defn- alternate-possible
198    "Where `possibles` is a list of characters which have been marked `possible` in these
199     `patterns`, return the first which has never been marked as `possible` in this
200     `position`.
201  
202     As with `generate`, which depends on it, this is experimental code which is currently
203     not used."
204    [patterns possibles others position]
205    (let [already-tried (remove nil?
206                                (map #(let [e (nth % position)]
207                                        (when (= :not-present (first e))
208                                          (nth e 1)))
209                                     patterns))
210          possibles'    (remove-all already-tried possibles)]
211      (if (empty? possibles')
212        (first others)
213        (first possibles'))))

215  (defn generate
216    "Generate the next word to test, given these `patterns` from previous tests, 
217     and these `candidates` for the word to be found. Note that the best word to 
218     test is not necessarily a member of `candidates`.
219  
220     TODO: the idea hear was to generate the most efficient possible test strings,
221     but it added a lot of complexity and doesn't work well so it is currently not
222     used."
223    [patterns candidates]
224    (let [f  (with-mark patterns :found)
225          fs (remove empty? f)
226          p  (seq (set (remove-all fs (all-with-mark patterns :present))))
227          o  (remove-all-with-mark patterns :not-present)]
228      (let [eureka (= (count fs) word-length)]
229        {:eureka eureka
230         :cand   (apply
231                  str
232                  (if eureka
233                    fs
234                    (map
235                     #(cond
236                        (nil? (nth f %)) (nth o %) ;; if it's a found position, try the next char we don't know about
237                        (> (count p) 1) (alternate-possible patterns p o %)
238                        :else (nth o %)) ;; not ideal; probably use `loop` instead of `map` 
239                            ;; so we can try others in strict order
240                     (range word-length))))})))

242  (defn ^:export solve
243    "Solve a wordle; with no arguments, start with a new random word; otherwise,
244     arguments are:
245     * `game` a function of one argument, as generated by `wordle-gen`, which takes
246       a guessed word as argument and returns a pattern;
247     * `patterns` a list of patterns from prior attempts;
248     * `candidates` a list of words to use as candidates for guessing."
249    ([] (wordle-gen))
250    ([game]
251     (solve game nil))
252    ([game patterns]
253     (solve game patterns words))
254    ([game patterns candidates]
255     (loop [i        0
256            to-test  (first candidates)
257            patterns patterns
258            cands    candidates]
259       (let [εὕρηκα (with-mark patterns :found)]
260         (cond (> i 6) nil
261               (every? char? εὕρηκα) {:word     (apply str εὕρηκα)
262                                      :attempts (count patterns)
263                                      :patterns (reverse patterns)}
264               :else (case (count cands)
265                       0 nil ;; fail
266                       1 {:word     (first cands)
267                          :attempts (count patterns)
268                          :patterns (reverse patterns)}
269         ;; else
270                       (let [pattern   (apply game (list to-test))
271                             patterns' (cons pattern patterns)
272                             cands'    (refine-candidates cands to-test pattern)]
273                         (recur (inc i)
274                                (first cands')
275                                patterns'
276                                (rest cands')))))))))
